(3)
SELECT *, COUNT(Votes.politicianId) as totalVote FROM Politicians
JOIN Votes
ON Politicians.id = Votes.politicianId
	WHERE name LIKE "Adam%"
	GROUP BY Politicians.id

(4)
SELECT COUNT(Votes.politicianId) AS totalVote, name, party, location  FROM Politicians
JOIN Votes
ON Politicians.id = Votes.politicianId
	GROUP BY name
	ORDER BY totalVote DESC
	LIMIT 3

ON Voters.id = Votes.voterId
